\begin{definitions}
	\definition{Трансляционные модели}{это генерация музыкальных произведений на основе немузыкальных данных, таких как графические образы или текст. Процесс может быть случайным или основываться на определенных правилах мелодической структуры, с использованием нейросетевых технологий для преобразования и распознавания исходных данных \cite{translation}.}
	
	\definition{Трансляция изображений}{это процесс генерации звуков на основе визуальной информации, представленной на изображении. \cite{alg}}
\end{definitions}
